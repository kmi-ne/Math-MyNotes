\chapter{Class algebra}

\section{Class}


\begin{dfn}[][][$x$ is a set/proper class]
  \begin{thmlist}
    \item
    \mylabel{Dfn_M}
    $
      \M{x} \defn ∃ y \ (x ∈ y)
    $
    \item
    \mylabel{Dfn_Pr}
    $
      \Pr{x} \defn ¬ \M{x}
    $
  \end{thmlist}
\end{dfn}


\begin{prp}[Thm_elem_is_set]
  $
    x ∈ y \ra \M{x}
  $
  \tcblower
  \begin{prf}
    \item
    \Show $\M{x}$
    \By{rcmrq, Dfn_M}
      \item
      \label{rcmrq}
      \T $∃ y \ (x ∈ y)$
      \By{2qwn5}
    \item
    \label{2qwn5}
    \Assume{$x ∈ y$}
  \end{prf}
\end{prp}


\begin{axm}[Axm_ext][][Axiom of Extensionality]
  $
    ∀ u \ (u ∈ x \lra u ∈ y) \ra x = y
  $
\end{axm}


\begin{dfn}[][][$x$ is a subclass/proper subclass of $y$]
  \begin{thmlist}
    \item
    \mylabel{Dfn_Sbc}
    $
      x ⊆ y \defn ∀ u \ (u ∈ x \ra u ∈ y)
    $
    \item
    \mylabel{Dfn_Psbc}
    $
      x ⊂ y \defn x ⊆ y ∧ x ≠ y
    $
  \end{thmlist}
\end{dfn}


\begin{prp}
  \begin{thmlist}
    \item
    \mylabel{Thm_Sbc_is_reflRel}
    $
      x ⊆ x
    $
    \item
    \mylabel{Thm_Sbc_is_transRel}
    $
      x ⊆ y ⊆ z \ra x ⊆ z
    $
    \item
    \mylabel{Thm_Sbc_is_antisymRel}
    $
      x ⊆ y ⊆ x \ra x = y
    $
  \end{thmlist}
  \tcblower
  \begin{thmlist}
    \item
    \begin{prf}
      \item
      $x ⊆ x$
      \By{a3b3n, Dfn_M}
      \item
      $∀ u \ (u ∈ x \ra u ∈ x)$
      \label{a3b3n}
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $x ⊆ z$
      \By{q8oe8}
        \item
        \label{q8oe8}
        \T $∀ u \ (u ∈ x \ra u ∈ z)$
        \By{04lc9}
        \item
        \label{04lc9}
        \T $
          \cuc{
            ∀ u \ (u ∈ x \ra u ∈ y) \\
            ∀ u \ (u ∈ y \ra u ∈ z)
          }
        $
        \By{8hjjr}
      \item
      \label{8hjjr}
      \Assume{$x ⊆ y ⊆ z$}
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $x = y$
      \By{7lu34, Axm_ext}
        \item
        \label{7lu34}
        \T $∀ u \ (u ∈ x \lra u ∈ y)$
        \By{0fh49}
        \item
        \label{0fh49}
        \T $
          \cuc{
            ∀ u \ (u ∈ x \ra u ∈ y) \\
            ∀ u \ (u ∈ y \ra u ∈ x)
          }
        $
        \By{d0fkc, Dfn_Sbc}
      \item
      \label{d0fkc}
      \Assume{$x ⊆ y ⊆ x$}
    \end{prf}
  \end{thmlist}
\end{prp}


\begin{prp}
  \begin{thmlist}
    \item \mylabel{Thm_Psbc_irreflRel}
    $
      ¬ (x ⊂ x)
    $
    \item \mylabel{Thm_psbc_has_less_elem}
    $
      x ⊂ y \lra
      \cuc{
        x ⊆ y \\
        ∃ u \ (u ∈ y ∧ u ∉ x)
      }
    $
    \item \mylabel{Thm_psbc_of_sbc_is_psbc}
    $
      x ⊂ y ⊆ z \ra x ⊂ z
    $
    \item \mylabel{Thm_sbc_of_psbc_is_psbc}
    $
      x ⊆ y ⊂ z \ra x ⊂ z
    $
    \item \mylabel{Thm_psbc_of_psbc_is_psbc}
    $
      x ⊂ y ⊂ z \ra x ⊂ z
    $
    \item \mylabel{Thm_sbc_eqv_psbc_or_Eq}
    $
      x ⊆ y \lra x ⊂ y ∨ x = y
    $
  \end{thmlist}
  \tcblower
  \begin{thmlist}
    \item
    \begin{prf}
      \item
      $¬ (x ⊂ x)$
      \By{5cn7h, Dfn_Psbc}
      \item
      \label{5cn7h}
      $¬ (x ⊆ x ∧ x ≠ x)$
      \By{i1ilq}
      \item
      \label{i1ilq}
      $x = x$
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $
        \rx{\cuc{
          x ⊆ y \\
          ∃ u \ (u ∈ y ∧ u ∉ x)
        }}
        \ra x ⊂ y
      $
        \item
        \T \Show $x ⊂ y$
        \By{mo0ms, m75gb}
          \item
          \label{m75gb}
          \T\T \Show $x ≠ y$
          \By{xrjl8, bvdxl}
            \item
            \label{xrjl8}
            \T\T\T $x = y \ra ¬ ∃ u \ (u ∈ y ∧ u ∉ x)$
        \item
        \label{mo0ms}
        \T \Assume{$x ⊆ y$}
        \item
        \label{bvdxl}
        \T \Assume{$∃ u \ (u ∈ y ∧ u ∉ x)$}
      \item
      \Show $
        \rx{x ⊂ y} \ra
        \cuc{
          x ⊆ y \\
          ∃ u \ (u ∈ y ∧ u ∉ x)
        }
      $
        \item
        \T \Show $∃ u \ (u ∈ y ∧ u ∉ x)$
        \By{lie67, Dfn_Sbc}
          \item
          \label{lie67}
          \T\T $¬ (y ⊆ x)$
          \By{i0gyn, gf8fg, Thm_Sbc_is_antisymRel}
          \item
          \label{gf8fg}
          \T\T $x ≠ y$
          \By{w810y, Dfn_Psbc}
        \item
        \label{i0gyn}
        \T \Show $x ⊆ y$
        \By{w810y, Dfn_Psbc}
        \item
        \label{w810y}
        \T \Assume{$x ⊂ y$}
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $x ⊂ z$
      \By{bjai4, xmb4n, Thm_psbc_has_less_elem}
        \item
        \label{bjai4}
        \T \Show $x ⊆ z$
        \By{o2zw9, Thm_Sbc_is_transRel}
          \item
          \label{o2zw9}
          \T\T $x ⊆ y ⊆ z$
          \By{zko72, Dfn_Psbc}
        \item
        \label{xmb4n}
        \T \Show $∃ u \ (u ∈ z ∧ u ∉ x)$
        \By{slopv, nor9n}
          \item
          \label{nor9n}
          \T\T $u ∈ y \ra u ∈ z$
          \By{zko72, Dfn_Sbc}
          \item
          \label{slopv}
          \T\T $∃ u \ (u ∈ y ∧ u ∉ x)$
          \By{zko72, Thm_psbc_has_less_elem}
      \item
      \label{zko72}
      \Assume{$x ⊂ y ⊆ z$}
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $x ⊂ z$
      \By{94hb6, novoe, Thm_psbc_has_less_elem}
        \item
        \label{94hb6}
        \T \Show $x ⊆ z$
        \By{5jt5y, Thm_Sbc_is_transRel}
          \item
          \label{5jt5y}
          \T\T $x ⊆ y ⊆ z$
          \By{naqn1, Dfn_Psbc}
        \item
        \label{novoe}
        \T \Show $∃ u \ (u ∈ z ∧ u ∉ x)$
        \By{9emdq, 4im8d}
          \item
          \label{4im8d}
          \T\T $u ∈ x \ra u ∈ y$
          \By{naqn1, Dfn_Sbc}
          \item
          \label{9emdq}
          \T\T $∃ u \ (u ∈ z ∧ u ∉ y)$
          \By{naqn1, Thm_psbc_has_less_elem}
      \item
      \label{naqn1}
      \Assume{$x ⊆ y ⊂ z$}
    \end{prf}
    \item
    \begin{prf}
      \item
      \Show $x ⊂ z$
      \By{ifxfr, Thm_psbc_of_sbc_is_psbc}
        \item
        \label{ifxfr}
        \T $x ⊂ y ⊆ z$
        \By{1vtkz, Dfn_Psbc}
      \item
      \label{1vtkz}
      \Assume{$x ⊂ y ⊂ z$}
    \end{prf}
    \item
    \begin{prf}
      \item
      $x ⊆ y \lra x ⊂ y ∨ x = y$
      \By{f117i, 60v5g}
        \item
        \label{60v5g}
        \T \Show $\sqc{x ⊆ y ∧ x ≠ y \\ x ⊆ y ∧ x = y} \lra x ⊂ y ∨ x = y$
        \By{z3e3y, qc1h8}
          \item
          \label{z3e3y}
          \T\T \Show $x ⊆ y ∧ x ≠ y \lra x ⊂ y$
          \By{Dfn_Psbc}
          \item
          \label{qc1h8}
          \T\T \Show $x ⊆ y ∧ x = y \lra x = y$
          \By{mu0eu, zimip}
            \item
            \label{zimip}
            \T\T\T \Show $x ⊆ y ∧ x = y \ra x = y$
            \item
            \label{mu0eu}
            \T\T\T \Show $x = y \ra x ⊆ y ∧ x = y$
            \By{Thm_Sbc_is_reflRel}
        \item
        \label{f117i}
        \T\Show $x ⊆ y \lra \sqc{x ⊆ y ∧ x ≠ y \\ x ⊆ y ∧ x = y}$
    \end{prf}
  \end{thmlist}
\end{prp}
