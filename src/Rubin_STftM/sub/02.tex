\begin{dfn}[][Unordered pair/unit set]
  \begin{thmlist}
    \item
    \mylabel{Dfn_pair}
    $
    \{x, y\} ≔ \cls{u}{u = x ∨ u = y}
    $
    \item
    \mylabel{Dfn_unit}
    $
      \{x\} ≔ \cls{u}{u = x}
    $
  \end{thmlist}
\end{dfn}


\begin{prp}
  $
    \{x, x\} = \{x\}
  $

  \step{

  }

\begin{prp}
  \begin{thmlist}
    \item
    $
      x ∈ \{x, y\}
    $
    \item
    $
      y ∈ \{x, y\}
    $
    \item

\begin{axm}[][][Axiom of Pairing]
  $
    \M{x} ∧ \M{y} \ra \M{\{x, y\}}
  $
\end{axm}


\begin{prp}
  $
    \M{x} \ra \M{\{x\}}
  $
\end{prp}


\begin{dfn}[][Ordered pair]
  $
    ⟨ x, y ⟩ ≔ \{ \{x\}, \{x, y\} \}
  $
\end{dfn}


\begin{thm}
  $
    \cuc{
      \M{x_𝟏} ∧ \M{x_𝟐} ∧ \M{y_𝟏} ∧ \M{y_𝟐} \\
      ⟨ x_𝟏, x_𝟐 ⟩ = ⟨ y_𝟏, y_𝟐 ⟩
    }
    \ra
    \cuc{
      x_𝟏 = y_𝟏 \\
      x_𝟐 = y_𝟐
    }
  $
\end{thm}


\begin{prp}
  $
    \M{x} ∧ \M{y} \ra \M{⟨ x, y ⟩}
  $
\end{prp}
